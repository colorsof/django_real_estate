What is eukaryotes

Eukaryotes are organisms whose cells have a nucleus - a membrane-bound compartment that contains their DNA.

ELI5 (Like you're 5):
Imagine a cell is like a house. In eukaryote cells, the DNA (the instruction manual) is kept in a special locked room (the nucleus). In prokaryote cells, the instruction manual is just lying around in the main room.

ELI10 (Like you're 10):
Eukaryotes are living things made of cells that have organized compartments. The most important compartment is the nucleus, which stores all the genetic information. It's like having different rooms in a house - kitchen, bedroom, bathroom - instead of one big open space. This organization lets the cell do more complex things.

Developer/Scientific Explanation:
Eukaryotes are one of the three domains of life (alongside Bacteria and Archaea). Key characteristics:

Structural features:

Nucleus - membrane-enclosed DNA
Membrane-bound organelles - mitochondria, endoplasmic reticulum, Golgi apparatus, etc.
Linear chromosomes (vs circular in prokaryotes)
Larger and more complex than prokaryotic cells
Examples:

Animals (including humans)
Plants
Fungi
Protists (algae, amoebas, etc.)
vs Prokaryotes:

Prokaryotes (bacteria, archaea) have NO nucleus
Their DNA floats freely in the cytoplasm
Much simpler internal structure
Generally smaller cells
Why this matters biologically: The nucleus and compartmentalization allow eukaryotic cells to regulate gene expression more precisely, enabling the complexity needed for multicellular organisms like us.

# businesses/models.py

from django.contrib.auth.models import User
from django.db import models
from django.utils import timezone

class Business(models.Model):
    """Each salon/business"""
    user = models.OneToOneField(User, on_delete=models.CASCADE)
    name = models.CharField(max_length=200)
    phone = models.CharField(max_length=15)
    created_at = models.DateTimeField(auto_now_add=True)
    
    def __str__(self):
        return self.name

class Customer(models.Model):
    """Salon's customers"""
    business = models.ForeignKey(Business, on_delete=models.CASCADE, related_name='customers')
    name = models.CharField(max_length=200)
    phone = models.CharField(max_length=15)
    last_visit = models.DateField(null=True, blank=True)
    sms_frequency_days = models.IntegerField(default=30)
    notes = models.TextField(blank=True)  # Optional notes about customer
    created_at = models.DateTimeField(auto_now_add=True)
    
    def __str__(self):
        return f"{self.name} - {self.business.name}"

class MessageTemplate(models.Model):
    """SMS message templates per salon"""
    business = models.ForeignKey(Business, on_delete=models.CASCADE, related_name='templates')
    name = models.CharField(max_length=100)  # "Appointment Reminder"
    message = models.TextField()  # "Hi {name}, time for your haircut at {business}!"
    is_active = models.BooleanField(default=True)
    created_at = models.DateTimeField(auto_now_add=True)
    
    def __str__(self):
        return f"{self.name} - {self.business.name}"

class SMSLog(models.Model):
    """Track all SMS sent"""
    customer = models.ForeignKey(Customer, on_delete=models.CASCADE, related_name='sms_logs')
    template = models.ForeignKey(MessageTemplate, on_delete=models.SET_NULL, null=True, blank=True)
    message = models.TextField()  # Actual message sent
    sent_at = models.DateTimeField(auto_now_add=True)
    status = models.CharField(max_length=20, choices=[
        ('sent', 'Sent'),
        ('delivered', 'Delivered'),
        ('failed', 'Failed'),
    ], default='sent')
    
    def __str__(self):
        return f"SMS to {self.customer.name} on {self.sent_at}"

class CheckIn(models.Model):
    """Manual check-in when customer visits"""
    customer = models.ForeignKey(Customer, on_delete=models.CASCADE, related_name='checkins')
    checked_in_at = models.DateTimeField(default=timezone.now)
    notes = models.TextField(blank=True)  # Optional: "Haircut + nails"
    
    def __str__(self):
        return f"{self.customer.name} checked in on {self.checked_in_at}"

class MPesaPayment(models.Model):
    """Track M-Pesa payments"""
    customer = models.ForeignKey(Customer, on_delete=models.CASCADE, related_name='payments')
    amount = models.DecimalField(max_digits=10, decimal_places=2)
    transaction_id = models.CharField(max_length=50, unique=True)  # M-Pesa receipt
    paid_at = models.DateTimeField()
    phone_number = models.CharField(max_length=15)  # Who paid
    created_at = models.DateTimeField(auto_now_add=True)
    
    def __str__(self):
        return f"{self.customer.name} paid KES {self.amount}"

class Business(models.Model):
    """Each business - Maloy Salon, DC Energy, etc."""
    user = models.OneToOneField(User, on_delete=models.CASCADE)
    name = models.CharField(max_length=200)
    phone = models.CharField(max_length=15)
    business_type = models.CharField(max_length=50, choices=[
        ('salon', 'Salon'),
        ('spa', 'Spa'),
        ('garage', 'Auto Service'),
        ('restaurant', 'Restaurant'),
        ('clinic', 'Clinic'),
    ])
    created_at = models.DateTimeField(auto_now_add=True)
    
    def __str__(self):
        return self.name

class BusinessCustomer(models.Model):
    """Junction table - tracks relationship between business and customer"""
    business = models.ForeignKey(Business, on_delete=models.CASCADE, related_name='business_customers')
    customer = models.ForeignKey(Customer, on_delete=models.CASCADE, related_name='businesses')
    
    # Relationship-specific fields
    last_visit = models.DateField(null=True, blank=True)
    sms_frequency_days = models.IntegerField(default=30)
    notes = models.TextField(blank=True)  # Business-specific notes about this customer
    is_active = models.BooleanField(default=True)  # Active relationship with THIS business
    created_at = models.DateTimeField(auto_now_add=True)
    
    class Meta:
        unique_together = ['business', 'customer']  # Can't add same customer twice
    
    def __str__(self):
        return f"{self.customer.name} at {self.business.name}"

from threading import local

_thread_locals = local()

def get_current_business():
    """Call this anywhere to get current business"""
    return getattr(_thread_locals, 'business', None)

class TenantMiddleware:
    def __init__(self, get_response):

        # Store business before request processing
        if request.user.is_authenticated:
